 33    To use the API, the user must have a valid certificate and a password.
 34    The password should be stored as an environment variable.
 35    
 36    Example usage:
 37    ```python
 38    scb = SCBapi()
 39    resultWrapper = api.up_to({"Antal arbetsställen": "4"}).sni(["62010"]).fetch().json()
 40    print(resultWrapper.json())
 41    ```

 43    :param cert_path: the path to the certificate file.

 45    def __init__(self, cert_path=CERT_PATH) -> None:
 46        self.api_base = 'https://privateapi.scb.se/nv0101/v1/sokpavar'
 47        self.api_pass = os.getenv("SCB_API_PASS")
 48        self.cert_path = cert_path
 49        self.owned_vars_from_api = {}
 50        self.variables_from_api = {}
 51        self.operator_map = {
 52            'Innehaller':self.contains,
 53            'ArLikaMed':self.equals,
 54            'BorjarPa':self.prefix,
 55            'FranOchMed':self.start_from,
 56            'TillOchMed':self.up_to,
 57            'Mellan':self.between,
 58            'Finns':self.exists,
 59            'FinnsInte':self.exists
 60        }
 61        self.default_json = {
 62            "Företagsstatus":"1",
 63            "Registreringsstatus":"1",
 64            "variabler": [],
 65            "Kategorier": []
 66        }
 67        self.json = {}

 73    def get_session(self):
 74        """
 75        Initializes a session with the SCB API.
 76        returns: a session object.
 77        """
 78        if not hasattr(self, 'session'):
 79            with Session() as s:
 80                s.mount(self.api_base, Pkcs12Adapter(pkcs12_filename=self.cert_path, pkcs12_password=self.api_pass))
 81            self.session = s
 82        else :
 83            self.session.close()
 84            with Session() as s:
 85                s.mount(self.api_base, Pkcs12Adapter(pkcs12_filename=self.cert_path, pkcs12_password=self.api_pass))
 86            self.session = s

107        """
108        Update the dictionary of all variables.

125    def update_owned_vars(self):
126        """
127        Update the list of owned variables.
128        The dictionary is in the format:
129            {'variable_name':[function name]}
130            Where the function names are:
131                'contains',                       # Innehaller
132                'equals',                         # ArLikaMed
133                'prefix',                         # BorjarPa
134                'starting' ( at incl.),           # FranOchMed
135                'ending' (at incl.),              # TillOchMed
136                'between'                         # Mellan
137                'exists'                          # Finns, FinnsInte
138        """
139        if not self.variables_from_api:
140            self.update_variables()
141
142        owned_vars_list = []
143        r = self._get_request("api/Je/KoptaVariabler").json()
144        for var in r['Variabler']:
145            owned_vars_list.append(var['Id_Variabel_JE'])
146        
147        self.owned_vars_from_api = {}
148        for var in owned_vars_list:
149            operators = self.variables_from_api[var]
150            self.owned_vars_from_api[var] = [self.operator_map[o].__name__ for o in operators]

371    def fetch_data(self, r_address, body=None, retries=1):
372        """
373        General method for creating requests against SCB API. Recursively retries the request if it fails for number of retries specified.
374        
375        :param r_address: the suffix of the address of the specific request.
376        :param body: the body to be sent with POST requests
377        :param retries: the number of retries to be made if the request fails.
378        :returns: a response object
379        """
380        
381        if (body is None):   
382            r = self.session.get(f'{self.api_base}/{r_address}') # En request
383        else :
384            r = self.session.post(f'{self.api_base}/{r_address}', json=body)
385        
386        if r.status_code != 200:
387            logging.error("Request failed with status code %s", r.status_code)
388            if body is not None:
389                logging.error("Request Body: %s", body)
390            logging.error("Response Text: %s", r.text)
391            logging.error("====================")
392            if (retries == 0):
393                sys.exit("Retries exhausted, request failed!")
394            logging.error("Retrying request...")
395            self.get_session()
396            r = self.fetch_data(r_address, body=body, retries=(retries - 1))
397        
398        return r

421    def _check_if_allowed(self, operator, var_list= None, var_dict = None):
422        """
423        Checks if the given variables are supported by the caller operation.
424
425        :param operator: a string (the exact name of the caller function).
426        :var_list: a list of variables used in the operation, or None.
427        :var_dict: a dictionary used in the operation, or None.
428
429        :raises VariableDoesNotExist: if the variable does not appear in the variable list fetched from the API.
430        :raises DoesNotOwnVariable: if the user does not own the chosen variable.
431        :raises VariableDoesNotSupportOperation: if the variable does not support the chosen operation (the caller).
432        """
433        if var_list is not None:
434            for var in var_list:
435                if var not in self.variables_from_api:
436                    raise VariableDoesNotExist("'%s' does not exist in the list of all variables!" % (var))
437                if var not in self.owned_vars_from_api:
438                    raise DoesNotOwnVariable("'%s' exists, but is not in the list of owned variables!" % (var))
439                if operator not in self.owned_vars_from_api[var]:
440                    raise VariableDoesNotSupportOperation("'%s' does not support '%s'!" % (var, operator))
441
442        if var_dict is not None:
443            for var in var_dict.keys():
444                if var not in self.variables_from_api:
445                    raise VariableDoesNotExist("'%s' does not exist in the list of all variables!" % (var))
446                if var not in self.owned_vars_from_api:
447                    raise DoesNotOwnVariable("'%s' exists, but is not in the list of owned variables!" % (var))
448                if operator not in self.owned_vars_from_api[var]:
449                    raise VariableDoesNotSupportOperation("'%s' does not support '%s'!" % (var, operator))

422    def _check_if_allowed(self, operator, var_list= None, var_dict = None):
423        """
424        Checks if the given variables are supported by the caller operation.
425
426        :param operator: a string (the exact name of the caller function).
427        :var_list: a list of variables used in the operation, or None.
428        :var_dict: a dictionary used in the operation, or None.
429
430        :raises VariableDoesNotExist: if the variable does not appear in the variable list fetched from the API.
431        :raises DoesNotOwnVariable: if the user does not own the chosen variable.
432        :raises VariableDoesNotSupportOperation: if the variable does not support the chosen operation (the caller).
433        """
434        if var_list is not None:
435            for var in var_list:
436                if var not in self.variables_from_api:
437                    raise VariableDoesNotExist("'%s' does not exist in the list of all variables!" % (var))
438                if var not in self.owned_vars_from_api:
439                    raise DoesNotOwnVariable("'%s' exists, but is not in the list of owned variables!" % (var))
440                if operator not in self.owned_vars_from_api[var]:
441                    raise VariableDoesNotSupportOperation("'%s' does not support '%s'!" % (var, operator))
442
443        if var_dict is not None:
444            for var in var_dict.keys():
445                if var not in self.variables_from_api:
446                    raise VariableDoesNotExist("'%s' does not exist in the list of all variables!" % (var))
447                if var not in self.owned_vars_from_api:

449                if operator not in self.owned_vars_from_api[var]:
450                    raise VariableDoesNotSupportOperation("'%s' does not support '%s'!" % (var, operator))
